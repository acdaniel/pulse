1 /**
  2  * Copyright 2011 Paranoid Ferret Productions
  3  *
  4  * The base type for all objects added to the world.
  5  * @param {object} [params] parameters that can be set as initialized options
  6  * on the node
  7  * @config {string} [name] name of the node
  8  * @class The base node object.
  9  */
 10 
 11 PFPlay.Node = Class.extend(
 12 /** @lends PFPlay.Node.prototype */
 13 {
 14   /** @constructs */
 15   init: function(params) {
 16 
 17     params = PFPlay.util.checkParams(params, {
 18       name : "Node" + PFPlay.Node.nodeIdx++
 19     });
 20 
 21     // Default the name to something unique.
 22     this.name = params.name;
 23 
 24     /** 
 25      * Private properties of the visual node. Should not need or use these.
 26      * @type {object}
 27      */
 28     this._private = { };
 29   },
 30 
 31   /**
 32    * The name of the node.
 33    * @type {string}
 34    */
 35   name : null,
 36 
 37   /**
 38    * Update function called on each loop in the engine
 39    * @param {number} elapsed the elapsed time since last call in 
 40    * milliseconds
 41    */
 42   update : function(elapsed) {
 43     // Nothing is updated by default
 44     if(PFPlay.DEBUG) {
 45       console.log("node update");
 46     }
 47   }
 48 
 49 });
 50 
 51 // Static index that's incremented whenever a node is created.
 52 // Used for uniquely naming nodes if a name is not specified.
 53 PFPlay.Node.nodeIdx = 0;
