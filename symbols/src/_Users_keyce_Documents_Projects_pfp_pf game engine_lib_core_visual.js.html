1 /**
  2  * Copyright 2011 Paranoid Ferret Productions
  3  *
  4  * The base type for all visual elements added to the world.
  5  * @param {object} [params] parameters that can be set as initialized options
  6  * on the node
  7  * @config {string} [name] name of the node
  8  * @author Charlie Key
  9  * @class visual node
 10  * @augments PFPlay.Node
 11  */
 12  
 13 PFPlay.Visual = PFPlay.Node.extend(
 14 /** @lends PFPlay.Visual.prototype */
 15 {
 16   /** @constructs */
 17   init : function(params) {
 18     this._super(params);
 19 
 20     /**
 21      * The html canvas element that represents this visual node.
 22      * @type {HTMLCanvas}
 23      */
 24     this.canvas = document.createElement('canvas');
 25 
 26     /** 
 27      * 2d context for the canvas element for this visual node.
 28      * @type {CanvasRenderingContext2D}
 29      */
 30     this._private.context = this.canvas.getContext('2d');
 31 
 32     /** 
 33      * Signifies it's the first update for the visual node.
 34      * @type {boolean}
 35      */
 36     this._private.firstUpdate = true;
 37 
 38     /** 
 39      * The current position of the visual node.
 40      * @type {point}
 41      */
 42     this.position = { 
 43       x : 0, 
 44       y : 0
 45     };
 46 
 47     /** 
 48      * The previous position of the visual node, previous position gets
 49      * updated in the update function.
 50      * @type {point}
 51      */
 52     this.positionPrevious = {
 53       x : 0,
 54       y : 0
 55     };
 56 
 57     /** 
 58      * The current size of the visual node.
 59      * @type {size}
 60      */
 61     this.size = {
 62       width : 0,
 63       height : 0
 64     };
 65 
 66     /** 
 67      * The previous size of the visual node, previous size gets
 68      * updated in the update function.
 69      * @type {size}
 70      */
 71     this.sizePrevious = {
 72       width : 0,
 73       height : 0
 74     };
 75 
 76     /**
 77      * The bounds of this visual node, updated based on size and position.
 78      * @type {rect}
 79      */
 80     this.bounds = {
 81       x : 0,
 82       y : 0,
 83       width : 0,
 84       height : 0
 85     };
 86 
 87     /**
 88      * The previous bounds of this visual node.
 89      * @type {rect}
 90      */
 91     this.boundsPrevious = {
 92       x : 0,
 93       y : 0,
 94       width : 0,
 95       height : 0
 96     };
 97 
 98     /**
 99      * The current position anchor of the visual node, the position on the 
100      * visual node sets the position of the anchor.
101      * @type {point}
102      */
103     this.anchor = {
104       x : 0.5,
105       y : 0.5
106     };
107 
108     /**
109      * The previous position anchor of the visual node, previous anchor gets 
110      * updated in the update function.
111      * @type {point}
112      */
113     this.anchorPrevious = {
114       x : 0.5,
115       y : 0.5
116     };
117 
118     /**
119      * The radius from the center of the visual node to the anchor point.
120      * @type {float}
121      */
122     this.anchorRadius = 0;
123 
124     /**
125      * The angle of the anchor to the center of the visual node in degrees.
126      * @type {float}
127      */
128     this.anchorAngle = 0;
129 
130     /** 
131      * The current scale of the visual node.
132      * @type {point}
133      */
134     this.scale = {
135       x : 1.0,
136       y : 1.0
137     };
138 
139     /** 
140      * The previous scale of the visual node, this is updated in the update
141      * function.
142      * @type {point}
143      */
144     this.scalePrevious = {
145       x : 1.0,
146       y : 1.0
147     };
148 
149     /**
150      * The rotation of the visual node, in degrees.
151      * @type {number}
152      */
153     this.rotation = 0;
154 
155     /**
156      * The previous rotation of the visual node, in degrees. This is updated in 
157      * the update function.
158      * @type {number}
159      */
160     this.rotationPrevious = 0;
161 
162     /**
163      * The position of the upper left of the visual node based on the 
164      * position and anchor.
165      * @type {point}
166      */
167     this.positionTopLeft = {
168       x : 0,
169       y : 0
170     };
171 
172     /**
173      * The previous position of the upper left of the visual node, previous
174      * top left gets updated in update function.
175      * @type {point}
176      */
177     this.positionTopLeftPrevious = {
178       x : 0,
179       y : 0
180     };
181 
182     /**
183      * This signifies whether the visual node needs to have certain size 
184      * properties recalculated.
185      * @type {boolean}
186      */
187     this.invalidProperties = true;
188 
189     /**
190      * The zindex of the visual node. Higher zindex means the visual object
191      * is on top of lower zindex nodes.
192      * @type {number}
193      */
194     this.zindex = 0;
195 
196     /**
197      * The previous zindex of the visual node. Higher zindex means the 
198      * visual object is on top of lower zindex nodes.
199      * @type {number}
200      */
201     this.zindexPrevious = 0;
202 
203     /**
204      * This signifies that the visual node has changed sort order.
205      * @type {boolean}
206      */
207     this.shuffled = false;
208 
209     /**
210      * The alpha transparency of the visual from 0-100.
211      * @type {number}
212      */
213     this.alpha = 100;
214 
215     /**
216      * The previous alpha transparency of the visual from 0-100.
217      * @type {number}
218      */
219     this.alphaPrevious = 100;
220 
221     /**
222      * This signifies whether the visual node is visible at the time, if false
223      * the node's draw function will not be called.
224      * @type {boolean}
225      */
226     this.visible = true;
227 
228     /**
229      * The previous state of visible, this value is updated in the 
230      * update function.
231      * @type {number}
232      */
233     this.visiblePrevious = true;
234 
235     /**
236      * This signifies whether the object needs to be redrawn on next draw phase.
237      * @type {boolean}
238      */
239     this.updated = true;
240   },
241 
242   /**
243    * Sets the previous position, used to clear the previous frame from
244    * a layer.
245    * @param {number} x the horizontal previous position of the sprite
246    * @param (number) y the vertical  previous position of the sprite
247    */
248   setPreviousPosition : function(x, y) {
249     this.positionPrevious = {
250       x : x,
251       y : y
252     }
253   },
254 
255   /**
256    * Moves the visual node by adding passed in parameters to the 
257    * current position.
258    * @param {number} x the value to add to the x position.
259    * @param {number} y the value to add to the y position.
260    */
261   move : function(x, y) {
262     this.position = {
263       x : this.position.x + x,
264       y : this.position.y + y
265     };
266   },
267 
268   /**
269    * Updates visual node properities by checking to see if they have 
270    * changed.
271    * @param {number} elapsed time elapsed since last update call in 
272    * milliseconds 
273    */
274   update : function(elapsed) {
275     // call the super update
276     this._super(elapsed);
277 
278     if(this._private.firstUpdate) {
279       this._private.firstUpdate = false;
280       this.invalidProperties = true;
281     }
282 
283     if(this.position.x != this.positionPrevious.x ||
284        this.position.y != this.positionPrevious.y) {
285       this.positionPrevious = this.position;
286       this.invalidProperties = true; 
287     }
288 
289     if(this.size.width != this.sizePrevious.width ||
290        this.size.height != this.sizePrevious.height) {
291       this.sizePrevious = this.size;
292       this.invalidProperties = true; 
293     }
294 
295     if(this.anchor.x != this.anchorPrevious.x ||
296        this.anchor.y != this.anchorPrevious.y) {
297       this.anchorPrevious = this.anchor;
298       this.invalidProperties = true;
299     }
300 
301     if(this.scale.x != this.scalePrevious.x ||
302        this.scale.y != this.scalePrevious.y) {
303       this.scalePrevious = this.scale;
304       this.invalidProperties = true;
305     }
306 
307     if(this.rotation != this.rotationPrevious) {
308       this.rotationPrevious = this.rotation;
309       this.invalidProperties = true;
310     }
311 
312     if(this.zindex != this.zindexPrevious) {
313       this.zindexPrevious = this.zindex;
314       this.shuffled = true;
315       this.updated = true;
316     }
317 
318     if(this.alpha != this.alphaPrevious) {
319       this.alphaPrevious = this.alpha;
320       this.updated = true;
321     }
322 
323     if(this.visible != this.visiblePrevious) {
324       this.visiblePrevious = this.visible;
325       this.updated = true;
326     }
327 
328     if(this.invalidProperties) {
329       this.calculateProperties();
330       this.updated = true; 
331     }
332   },
333 
334   /**
335    * Draws this visual node to passed in context. This is draw the canvas
336    * for this visual node on the context applying rotation, scale, and
337    * alpha.
338    * @param {CanvasRenderingContext2D} ctx the context in which to draw 
339    * visual on
340    */
341   draw : function(ctx) {
342     if(PFPlay.DEBUG) {
343       console.log('visual node draw');
344     }
345 
346     ctx.save();
347 
348     if(PFPlay.DEBUG) {
349       ctx.save();
350       ctx.fillStyle = "#CCDE42";
351       ctx.beginPath();
352       ctx.arc(
353         this.positionTopLeft.x + this.canvas.width / 2, 
354         this.positionTopLeft.y + this.canvas.height / 2, 
355         3, 0, Math.PI * 2, true
356       );
357       ctx.closePath();
358       ctx.fill();
359       ctx.restore();
360     } 
361 
362     if(PFPlay.DEBUG) {
363       ctx.strokeStyle = "#0022FF";
364       ctx.strokeRect(
365         this.positionTopLeft.x, this.positionTopLeft.y, 
366         this.canvas.width, this.canvas.height
367       );
368     }
369 
370     // apply the alpha for this visual node
371     ctx.globalAlpha = this.alpha / 100;
372 
373     // apply the rotation if needed
374     if(this.rotation > 0) {
375       var rotationX = this.positionTopLeft.x + 
376                       this.size.width * this.scale.x / 2;
377       var rotationY = this.positionTopLeft.y + 
378                       this.size.height * this.scale.y / 2;
379 
380       ctx.translate(rotationX, rotationY);
381       ctx.rotate((Math.PI * (this.rotation % 360)) / 180);
382       ctx.translate(-rotationX, -rotationY);
383     }
384 
385     // apply the scale
386     ctx.scale(this.scale.x, this.scale.y);
387 
388     // draw the canvas
389     ctx.drawImage(
390       this.canvas, 
391       this.positionTopLeft.x / this.scale.x, 
392       this.positionTopLeft.y / this.scale.y
393     );
394 
395     if(PFPlay.DEBUG) {
396       ctx.strokeStyle = "#22FF33";
397       ctx.strokeRect(
398         this.positionTopLeft.x / this.scale.x, 
399         this.positionTopLeft.y / this.scale.y,
400         this.canvas.width, 
401         this.canvas.height
402       );
403     }  
404     ctx.restore();
405 
406     if(PFPlay.DEBUG) {
407       ctx.save();
408       ctx.fillStyle = "#FF3300";
409       ctx.beginPath();
410       ctx.arc(this.position.x, this.position.y, 3, 0, Math.PI * 2, true);
411       ctx.closePath();
412       ctx.fill();
413       ctx.restore();
414     }
415 
416     this.updated = false;
417   },
418 
419   /**
420    * Calculates new anchor radius and angle based on the size of the 
421    * visual node and anchor position. It will then calculate a new top left
422    * position for this node by calculating canvas width and height based on
423    * the rotation of this node and position. Lastly, it sets the 
424    * width and height of this node's canvas.
425    */
426   calculateProperties : function() {
427 
428     // calculate new anchor radius based on size and scale
429     var sw = this.size.width;
430     var sh = this.size.height;
431     var sx = sw / 2, sy = sh / 2;
432     var ix = this.anchor.x * sw, iy = this.anchor.y * sh;
433     var dx = ix - sx, dy = iy - sy;
434     this.anchorRadius = Math.sqrt(dx * dx + dy * dy);
435     this.anchorAngle = Math.atan2(dy, dx) * 180 / Math.PI - 90;
436     if(isNaN(this.anchorAngle)) {
437      this.anchorAngle = 0;
438     }
439 
440     // save previous top left
441     this.positionTopLeftPrevious = this.positionTopLeft;
442 
443     var ox = this.size.width * this.scale.x / 2;
444     var oy = this.size.height * this.scale.y / 2;
445     var xpos = this.position.x - 
446                Math.sin(Math.PI * -(this.rotation + this.anchorAngle) / 180) * 
447                this.anchorRadius - ox;
448     var ypos = this.position.y - 
449                Math.cos(Math.PI * -(this.rotation + this.anchorAngle) / 180) * 
450                this.anchorRadius - oy;
451     
452     this.positionTopLeft = {
453       x : xpos, 
454       y : ypos
455     };
456 
457     // update the canvas element
458     this.canvas.width = this.size.width;
459     this.canvas.height = this.size.height;
460 
461     // update the bounds
462     this.boundsPrevious = this.bounds;
463     this.bounds = {
464       x : this.positionTopLeft.x,
465       y : this.positionTopLeft.y,
466       width : this.size.width * this.scale.x,
467       height : this.size.height * this.scale.y
468     };
469 
470     this.invalidProperties = false;
471   }
472 });
