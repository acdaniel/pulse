1 /**
  2  * Copyright 2011 Paranoid Ferret Productions
  3  * 
  4  * Animations are a collection of rectangles that represent pieces of
  5  * a larger rectangle, which change at a given rate, IE the frame
  6  * changes at a given interval.
  7  * @param {object} params parameters that can be set as initialized options
  8  * on the node
  9  * @config {PFPlay.Sprite} target the sprite for the action to be performed on
 10  * @config {string} [name] name of the node
 11  * @config {object} [size] a point object that represents the size of the 
 12  * animation frames
 13  * @config {number|array} [frames] a number representing the number of frames
 14  * or an array of frames representing the sequence in which to play the
 15  * frames
 16  * @config {number} [frameRate] the number of frames per second
 17  * @config {point} [offset] the pixel offset of the animation in respect to the 
 18  * sprite sheet
 19  * @config {size} [bounds] the phyiscal bounds in which all frames must lie
 20  * @config {number} [plays] the number of times the animation should loop, if 
 21  * less than 0 it will loop continuously
 22  * @author Richard Key
 23  * @class animation action
 24  * @augments PFPlay.Action
 25  */
 26  
 27 PFPlay.AnimateAction = PFPlay.Action.extend(
 28 /** @lends PFPlay.AnimateAction.prototype */
 29 {
 30   /** @constructs */ 
 31   init : function(params) {
 32 
 33     this._super(params);
 34 
 35     // Check the params 
 36     params = PFPlay.util.checkParams(params,
 37     {
 38       name : this.name,
 39       size : {width:0, height:0},
 40       frames : 0,
 41       frameRate : 0,
 42       offset : {x:0, y:0},
 43       bounds : {width:1, height:1},
 44       plays : -1
 45     });
 46 
 47     /** 
 48      * The size of the frame for the animation.
 49      * @type {size}
 50      */
 51     this.size = params.size;
 52 
 53     /** 
 54      * @private
 55      * The original frame of the texture when started.
 56      * @type {size}
 57      */
 58     this._private.frameOriginal = {
 59       x : 0,
 60       y : 0,
 61       width : 0,
 62       height : 0
 63     };
 64 
 65     /** 
 66      * @private
 67      * The bounds in which all frames must lie.
 68      * @type {size}
 69      */
 70     this._private.bounds = params.bounds;
 71 
 72     /**
 73      * @private
 74      * The number of frames in the sequence.
 75      * @type {number}
 76      */
 77     this._private.frames = params.frames;
 78 
 79     /**
 80      * @private
 81      * The frame rate that the animation should play at.
 82      * @type {number}
 83      */
 84     this._private.frameRate = params.frameRate;
 85 
 86     /** 
 87      * @private
 88      * The pixel offset for the frames on the original image.
 89      * @type {point}
 90      */
 91     this._private.offset = params.offset;
 92 
 93     /**
 94      * @private
 95      * The number of play throughs if it's equal to less than one it loops.
 96      * @type {number}
 97      */
 98     this._private.plays = params.plays;
 99 
100     /**
101      * @private
102      * The current play through number.
103      * @type {number}
104      */
105     this._private.currentPlay = 1;
106 
107     /**
108      * @private
109      * The current frame of the animation. Zero indexed.
110      * @type {number}
111      */
112     this._private.currentFrame = 0;
113 
114     /**
115      * @private
116      * The start time for the animation.
117      * @type {number}
118      */
119     this._private.start = 0;
120 
121     /**
122      * @private
123      * The current play time of the animation.
124      * @type {number}
125      */
126     this._private.playTime = 0;
127   },
128 
129   /** 
130    * Gets or Sets the current outer bounds for the animation. The bounds
131    * determine when to wrap frames to the next "line." Frames will always be
132    * forced to lie without the set bounds.
133    * @param {object} newBounds the new bounds of the animation
134    * @return {object} the current bounds of the animation 
135    */
136   bounds : function(newBounds)
137   {
138     if(newBounds == undefined || newBounds == null)
139       return this._private.bounds;
140   
141     /** Bounds have to be greater than 0 to prevent an infinite loop
142      * when checking for positions outside them. */
143     if(newBounds.x <= 0)
144       newBounds.x = 1;
145     
146     if(newBounds.y <= 0)
147       newBounds.y = 1;
148       
149     this._private.bounds = newBounds;
150   },
151 
152   /**
153    * Gets a frame of the animation based on the passed in index or
154    * the current frame.
155    * @param {number} index the frame to retrieve
156    * @return {object} a rectangle that represents the frame
157    */
158   getFrame : function(index) 
159   {
160     var frame = PFPlay.util.checkValue(index, this._private.currentFrame);
161     
162     if(this._private.frames instanceof Array) {
163       frame = this._private.frames[this._private.currentFrame];
164     }
165     
166     var x = (frame + this._private.offset.x) * this.size.width;
167     var y = this._private.offset.y;
168     
169     /** If the x position is out of bounds, move the frame selection
170      * down. */
171     while(x >= this._private.bounds.x)
172     {
173       x = x - this._private.bounds.x;
174       y++;
175     }
176       
177     y = y * this.size.height;
178     
179     if(y >= this._private.bounds.y) {
180       y = this._private.bounds.y - this.size.height;
181     }
182     
183     return {x: x, y: y, width: this.size.width, height: this.size.height};
184   },
185 
186   /**
187    * Starts the animation.
188    */
189   start : function() {
190     this._super();
191     this._private.currentPlay = 1;
192     this._private.frameOriginal = this.target.textureFrame;
193   },
194 
195   /**
196    * Pauses the animation.
197    */
198   pause : function() {
199     this._super();
200   },
201 
202   /**
203    * Stops the animation.
204    */
205   stop : function() {
206     this._super();
207     this._private.currentFrame = 0;
208     this._private.playTime = 0;
209   },
210 
211   /**
212    * Complete handler when the animation is complete.
213    */
214   complete : function() {
215     this._super();
216     this.target.textureFrame = this._private.frameOriginal;
217   },
218 
219   /** 
220    * This is the master update function for the animation. This will 
221    * update the target's texture with the correct frame.
222    * @param {number} elapsed the elapsed time since last update call in 
223    * milliseconds 
224    */
225   update : function(elapsed)
226   {
227     this._super();
228 
229     if(this.running == false)
230     {
231       return;
232     }
233 
234     var updated = false;
235 
236     this._private.playTime += elapsed;
237     
238     if(this._private.playTime >= (1 / this._private.frameRate * 1000))
239     {
240       this._private.currentFrame++;
241       
242       var length = this._private.frames.length || this._private.frames;
243 
244       if(this._private.currentFrame >= length) {
245         this._private.currentFrame = 0;
246         this._private.currentPlay++;
247         if(this._private.plays > 0 && 
248            this._private.currentPlay > this._private.plays) {
249           this.stop();
250           this.complete();
251         }
252       }
253       
254       this._private.playTime = 0;
255       updated = true;
256     }
257 
258     if(updated) {
259       // update the targets texture
260       var frame = this.getFrame();
261       this.target.textureFrame = frame;
262       this.target.updated = true;
263     }
264   }
265 });
