1 /**
  2  * Copyright 2011 Paranoid Ferret Productions
  3  *
  4  * Scenes are collections of layers that the basis of all
  5  * things visual in the engine.
  6  * @param {object} [params] parameters that can be set as initialized options
  7  * on the node
  8  * @config {string} [name] name of the node
  9  * @class scene class
 10  * @augments PFPlay.Node
 11  */
 12 
 13 PFPlay.Scene = PFPlay.Node.extend(
 14 /** @lends PFPlay.Scene.prototype */
 15 {
 16   /** @constructs */
 17   init : function(params) {
 18 
 19     // Call the super constructor
 20     this._super(params);
 21 
 22     /**
 23      * Associative array (object) of the layers for this scene.
 24      * @type {object}
 25      */
 26     this.layers = {};
 27 
 28     /** 
 29      * @private
 30      * Associative array (object) actual canvases for the layers.
 31      * @type {object}
 32      */
 33     this._private.liveLayers = {};
 34 
 35     /**
 36      * @private
 37      * Array of the ordered keys for the layers.
 38      * @type {array}
 39      */
 40     this._private.orderedKeys = [];
 41 
 42     /**
 43      * @private
 44      * The default size for layers
 45      * @type {size}
 46      */
 47     this._private.defaultSize = {
 48       width : 0,
 49       height : 0
 50     };
 51 
 52     /**
 53      * Whether this scene is active.
 54      * @type {boolean}
 55      */
 56     this.active = false;
 57 
 58     /** private reference to this */
 59     var _self = this;
 60     
 61     /**
 62      * The event manager for this scene, handles all events.
 63      * @type {PFPlay.EventManager}
 64      */
 65     this.events = new PFPlay.EventManager({masterCallback:
 66       function(type, evt)
 67       {
 68         for(var l in _self.layers)
 69         {
 70           var lBounds = _self.layers[l].bounds;
 71           
 72           if(PFPlay.events[type] == 'mouse')
 73           {          
 74             if(evt.world.x > lBounds.x && 
 75               evt.world.x < (lBounds.x + lBounds.width) && 
 76               evt.world.y > lBounds.y && 
 77               evt.world.y < (lBounds.y + lBounds.height))
 78             {
 79               // Adjust the event params based on the layer's position.
 80               evt.parent.x = evt.position.x;
 81               evt.parent.y = evt.position.y;
 82               evt.position.x = evt.world.x - lBounds.x;
 83               evt.position.y = evt.world.y - lBounds.y;
 84           
 85               _self.layers[l].events.raiseEvent(type, evt);
 86             }
 87           } else {
 88             _self.layers[l].events.raiseEvent(type, evt);
 89           }
 90             
 91         }
 92       }
 93     });
 94   },
 95 
 96   /**
 97    * Changes the default size of the scene. This will update all the
 98    * default layers sizes too.
 99    * @param {number} width the new default width
100    * @param {number} height the new default height
101    */
102   setDefaultSize : function(width, height) {
103     for(var l in this.layers)
104     {
105       if(this.layers[l].isDefaultSize == true) {
106         this.layers[l].resize(width, height); 
107       }
108     }
109       
110     this._private.defaultSize = {width: width, height: height};
111   },
112 
113   /**
114    * Adds a new layer to this scene.
115    * @param {PFPlay.Layer} layer to add
116    * @param {number} zindex the zindex of the new layer
117    */
118   addLayer : function(layer, zindex) {
119     if(layer instanceof PFPlay.Layer && 
120        !this.layers.hasOwnProperty(layer.name))
121     {
122       if(typeof zindex == 'number') {
123         layer.zindex = zindex;
124       }
125       
126       if(layer.isDefaultSize == true) {
127         layer.resize(
128           this._private.defaultSize.width, 
129           this._private.defaultSize.height
130         );
131       }
132       
133       this.layers[layer.name] = layer;
134       this._private.orderedKeys = PFPlay.util.getOrderedKeys(this.layers);
135     }
136   },
137   
138   /**
139    * Removes a layer from this scene by name.
140    * @param {string} name the name of the layer to remove
141    */
142   removeLayer : function(name) {
143     if(typeof name == 'string' && this.layers.hasOwnProperty(name)) {
144       delete _layers[name];
145     }
146   },
147   
148   /**
149    * Gets a layer from this scene by name.
150    * @param {string} name the name of the layer to get
151    */
152   getLayer : function(name) {
153     if(this.layers.hasOwnProperty(name)) {
154       return this.layers[name]; 
155     }
156     return null; 
157   },
158   
159   /**
160    * Gets a live canvas for a layer in this scene.
161    * @param {string} name the name of the layer 
162    */
163   getLiveLayer : function(name) {
164     if(this.layers.hasOwnProperty(name)) {
165       return this._private.liveLayers[name];
166     }
167     return null;
168   },
169 
170   /**
171    * Gets the container for this scene, a DOM div element.
172    * @return {DOMElement} returns the div container for this scene
173    */
174   getSceneContainer : function() {
175     var container = document.createElement('div');
176     container.id = this.name;
177     
178     for(var l = 0; l < this._private.orderedKeys.length; l++)
179     {
180       if(!this.layers[this._private.orderedKeys[l]]) {
181         continue;
182       }
183       
184       var layer = this.layers[this._private.orderedKeys[l]];
185       
186       var liveCanvas = layer.canvas.cloneNode(true);
187       liveCanvas.id = 'live:' + layer.name;
188       var ctx = liveCanvas.getContext('2d');
189 
190       this._private.liveLayers[layer.name] = ctx;
191       
192       container.appendChild(liveCanvas);
193     }
194     
195     return container;
196   },
197 
198   /**
199    * Updates the layers and then draws them.
200    * @param {number} elapsed time elapsed since last update call in 
201    * milliseconds 
202    */
203   update : function(elapsed) {
204 
205     this._super(elapsed);
206 
207     var reorder = false;
208     
209     /** Reorder the layers, if necessary, before drawing */
210     for(var l in this.layers)
211     {
212       if(this.layers[l].shuffled == true) {
213         reorder = true; 
214       }
215     }
216     
217     if(reorder == true) {
218       this._private.orderedKeys = PFPlay.util.getOrderedKeys(this.layers); 
219     }
220     
221     // update layers
222     for(var l in this.layers)
223     {
224       this.layers[l].update(elapsed);
225     }
226 
227     // draw layers
228     for(var o = 0; o < this._private.orderedKeys.length; o++) {
229       var layer = this.layers[this._private.orderedKeys[o]];
230 
231       if(layer.updated) {
232         var name = layer.name;
233         var ctx = this._private.liveLayers[layer.name];
234 
235         ctx.clearRect(0, 0, layer.size.width, layer.size.height);
236 
237         layer.draw(ctx);
238       }
239     }
240   }
241 
242 });
