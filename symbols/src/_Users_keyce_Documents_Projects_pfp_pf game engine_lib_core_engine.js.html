  8 PFPlay.Engine = function(params)
  9 {
 10   params = PFPlay.util.checkParams(params,
 11   {
 12     gameWindow: 'gameWindow',
 13     width: 0,
 14     height: 0
 15   });
 16   
 17   var _width = params.width;
 18   var _height = params.height;
 19   
 20   var _start = 0;
 21   var _gameWindow = null;
 22 
 23   //Create game window 
 24   if(typeof params.gameWindow == 'object') {
 25     _gameWindow = params.gameWindow;
 26   } else {
 27     _gameWindow = document.getElementById(params.gameWindow);
 28   }
 29   
 30   // Attempt to get the width from the specified container.
 31   if(_width == 0 || _height == 0) {
 32     if(_gameWindow) {
 33       var parentWidth = parseInt(_gameWindow.style.width);
 34       var parentHeight = parseInt(_gameWindow.style.height);
 35       
 36       if(parentWidth && parentHeight) {
 37         _width = parentWidth;
 38         _height = parentHeight;
 39       }
 40     }
 41   }
 42   
 43   // Width and height not found anywhere - default to something.
 44   if(_width == 0 || _height == 0) {
 45     _width = 640;
 46     _height = 480;
 47   }
 48   
 49   // Create a div within the supplied container to hold each layer.
 50   // This prevents issues based on the unknown parameters of the
 51   // user-supplied container element.
 52   var _mainDiv = document.createElement('div');
 53   _mainDiv.style.position = 'absolute';
 54   _mainDiv.style.width = _width + 'px';
 55   _mainDiv.style.height = _height + 'px';
 56   _mainDiv.style.overflow = 'hidden';
 57   _gameWindow.appendChild(_mainDiv);
 58   
 59   this.scenes = new PFPlay.SceneManager({gameWindow: _mainDiv});
 60   
 61   var _loopLogic = null;
 62 
 63   this.masterTime = 0;
 64   this.tick = 100;
 65 
 66   var _currentTime = new Date().getTime();
 67   var _lastTime = _currentTime;
 68   
 69   this.getWindowOffset = function()
 70   {
 71     var offX = _mainDiv.offsetLeft;
 72     var offY = _mainDiv.offsetTop;
 73     
 74     if(_mainDiv.offsetParent)
 75     {
 76       var parent = _mainDiv.offsetParent;
 77       do {
 78           offX += parent.offsetLeft;
 79           offY += parent.offsetTop;
 80       } while (parent = parent.offsetParent);
 81     }
 82     
 83     return {x: offX, y: offY};
 84   };
 85   
 86   this.bindEvents = function()
 87   {
 88     var eng = this;
 89     
 90     for(var e in PFPlay.events)
 91     {
 92       window.addEventListener(e, 
 93         function(evt) { eng.windowEvent(eng, evt) }, false);
 94     }
 95   };
 96   
 97   this.go = function(tick, loop)
 98   {
 99     this.tick = tick;
100     var eng = this;
101 
102     this.bindEvents();
103     
104     if(!loop)
105       _loopLogic = function() {};
106     else
107       _loopLogic = loop;
108     
109     setInterval(function() {eng.loop(eng);}, this.tick);
110   };
111   
112   this.loop = function(eng)
113   {
114     _currentTime = new Date().getTime();
115 
116     var elapsed = _currentTime - _lastTime;
117     
118     _loopLogic(eng.scenes);
119     
120     var activeScenes = eng.scenes.getScenes(true);
121     
122     for(var s = 0; s < activeScenes.length; s++)
123     {
124       activeScenes[s].update(elapsed);
125     }
126     
127     _lastTime = _currentTime;
128 
129     this.masterTime += elapsed;
130   };
131   
132   this.windowEvent = function(eng, rawEvt)
133   {    
134     var activeScenes = eng.scenes.getScenes(true);
135     var offset = eng.getWindowOffset();
136     
137     for(var s = 0; s < activeScenes.length; s++)
138     {
139       var x = rawEvt.clientX - offset.x + document.body.scrollLeft;
140       var y = rawEvt.clientY - offset.y + document.body.scrollTop;
141       
142       var evtProps = new PFPlay.MouseEvent();
143       evtProps.window.x = rawEvt.clientX;
144       evtProps.window.y = rawEvt.clientY;
145       evtProps.world.x = x;
146       evtProps.world.y = y;
147       evtProps.parent.x = evtProps.window.x;
148       evtProps.parent.y = evtProps.window.y;
149       evtProps.position.x = x;
150       evtProps.position.y = y;
151       
152       if(PFPlay.events[rawEvt.type] == 'keyboard')
153       {
154         var code;
155         if(rawEvt.charCode) code = rawEvt.charCode;
156         else if (rawEvt.keyCode) code = rawEvt.keyCode;
157         else if (rawEvt.which) code = rawEvt.which;
158         
159         evtProps['keyCode'] = code;
160         evtProps['key'] = String.fromCharCode(code);
161       }
162       
163       activeScenes[s].events.raiseEvent(rawEvt.type, evtProps);
164     }
165   };
166 }
