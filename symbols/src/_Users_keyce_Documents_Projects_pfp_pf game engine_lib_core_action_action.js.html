1 /**
  2  * The base class for actions.
  3  * @param {object} params parameters that can be set as initialized options
  4  * on the node
  5  * @config {PFPlay.Sprite} target the sprite for the action to be performed on
  6  * @config {string} [name] name of the node
  7  * @class base action
  8  * @augments PFPlay.Node
  9  */
 10 
 11 PFPlay.Action = PFPlay.Node.extend(
 12 /** @lends PFPlay.Action.prototype */
 13 {
 14   /**
 15    * Initilization function
 16    * @param {object} params for this action must include target
 17    * @constructs
 18    */ 
 19   init: function(params) {
 20     this._super(params);
 21       
 22     if(params.target == "") {
 23       throw "Target must be included for action.";
 24     }
 25       
 26     /**
 27      * The target node for the action.
 28      * @type {PFPlay.Node} 
 29      */
 30     this.target = params.target;
 31 
 32     /**
 33      * Whether this action is currently running or not.
 34      * @type {boolean}
 35      */
 36     this.isRunning = false;
 37 
 38     /**
 39      * Whether this action is completed or not.
 40      * @type {boolean}
 41      */
 42     this.isComplete = false;
 43 
 44     /**
 45      * Event manager for action. Used to raise events such as complete.
 46      * @type {PFPlay.EventManager}
 47      */
 48     this.events = new PFPlay.EventManager();
 49   },
 50 
 51   /**
 52    * Starts the action.
 53    */
 54   start: function() {
 55     if(PFPlay.DEBUG) {
 56       console.log('action started');
 57     }
 58     if(this.target.runningActions) {
 59       this.target.runningActions.push(this);
 60     }
 61     this.isComplete = false;
 62     this.isRunning = true;
 63   },
 64 
 65   /**
 66    * Pauses the action.
 67    */
 68   pause: function() {
 69     if(PFPlay.DEBUG) {
 70       console.log('action paused');
 71     }
 72     this.isRunning = false;
 73   },
 74 
 75   /**
 76    * Stops the action.
 77    */
 78   stop: function() {
 79     if(PFPlay.DEBUG) {
 80       console.log('action stopped');
 81     }
 82     this.isRunning = false;
 83   },
 84 
 85   /**
 86    * Called when the action is complete.
 87    */
 88   complete: function() {
 89     if(PFPlay.DEBUG) {
 90       console.log('action complete');
 91     }
 92     this.isComplete = true;
 93     this.events.raiseEvent('complete', {action : this});
 94   }
 95 });
